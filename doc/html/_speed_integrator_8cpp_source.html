34 void SpeedIntegrator::speedIntegrator()
35 {
36  if (!bUsrCtrl)
37  return;
38  QTimer::singleShot(readSpeedInterval, this, [=]() { speedIntegrator(); });
39  auto spd = plc->getTruckSpeed();
40  auto s_tmp = spd * readSpeedInterval / 60000;
41  for (auto& ckp : ckps) {
42  //从第一次触发后开始
43  if (!ckp.bFirstStart)
44  continue;
45  ckp.distance += s_tmp;
46  int ckpindex = &ckp - ckps;
47  if (!ckp.triBeginReady && ckp.distance > dist_min) {
48  emit expectTriBegin(ckpindex);
49  //qDebug() << "ckp expect begin" << ckpindex;
50  ckp.triBeginReady = true;
51  }
52  if (!ckp.triEndReady && ckp.distance > dist_max) {
53  emit expectTriEnd(ckpindex);
54  //qDebug() << "ckp expect end" << ckpindex;
55  ckp.triEndReady = true;
56  ckp.newWheel();
57  }
58  }
59 }
