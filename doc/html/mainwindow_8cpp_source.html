173 bool MainWindow::cleanDir(QString dirPath, int nDays)
174 {
175  //planA:only clean the file that auto generated by app
176  //planB:clean all the file in the directory, include auto-generated and other files, NOT recommended
177  bool rslt = true;
178 
179  QDir dir(dirPath);
180  //QStringList filters;
181  dir.setFilter(QDir::Dirs | QDir::NoDotAndDotDot); //prevent deleting the '.' and '..' dirs, very important
182  QFileInfoList fis = dir.entryInfoList();
183  auto today = QDate::currentDate();
184  for (const auto& fi : fis) {
185  //将文件名转换为日期
186  auto fn = fi.fileName();
187  auto fdt = QDate::fromString(fn, "yyyyMMdd");
188  if (fdt.isValid())//是否有效日期
189  {
190  if (fdt.daysTo(today) >= nDays)//如果到今天超过n天,则删除。default=0，即今天也删除
191  {
192  QDir selDir = fi.absoluteFilePath();
193  if (!selDir.removeRecursively())
194  rslt = false;
195  }
196  }
197  }
198  if (!rslt)
199  qDebug() << "cleanDir error";
200  return rslt;
201 }

435 void MainWindow::setupScheduler(int hh, int mm, bool execnow /*= true*/, bool everyday /*= true*/)
436 {
437  if (execnow) {
438  execScheduledTasks();
439  }
441  QDate date = QDate::currentDate();
442  if (QTime::currentTime().msecsTo(QTime(hh, mm, 00, 000)) <= 0) { //if the current time is after 12:00,
443  date = date.addDays(1); //2. than set date to tomorrow
444  }
445  auto ms_to_hhmm = QDateTime::currentDateTime().msecsTo(QDateTime(date, QTime(hh, mm))); //precision of 20ms
446  if (everyday) {
447  QTimer::singleShot(ms_to_hhmm, this, [=]() {
448  execScheduledTasks();
449  auto* timer24 = new QTimer(this);
450  connect(timer24, SIGNAL(timeout()), this, SLOT(execScheduledTasks()));
451  timer24->start(24 * 60 * 60 * 1000);
452  });
453  }
454  else {
455  QTimer::singleShot(ms_to_hhmm, this, SLOT(execScheduledTasks()));
456  }
457 }
