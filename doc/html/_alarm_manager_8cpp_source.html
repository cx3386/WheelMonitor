65 void AlarmManager::onHardwareAlarm(HardwareAlarmEvent ev)
66 {
67  switch (ev) {
68  case HardwareAlarmEvent::Reset:
69  currentCio100 = 0;
70  break;
71  case HardwareAlarmEvent::Outer_Warn:
72  currentCio100 |= 1 << 4; //CIO 100.04 out warn
73  break;
74  case HardwareAlarmEvent::Outer_Alarm:
75  currentCio100 |= 1 << 5; //CIO 100.05 out alarm
76  break;
77  case HardwareAlarmEvent::Inner_Warn:
78  currentCio100 |= 1 << 6; //CIO 100.06 in warn
79  break;
80  case HardwareAlarmEvent::Inner_Alarm:
81  currentCio100 |= 1 << 7; //CIO 100.07 in alarm
82  break;
83  default: // not do anything
84  qDebug() << "overflow";
85  break;
86  }
87  emit showCio100ToUi(currentCio100);
88  emit hardAlarmToPlc(currentCio100);
89  emit showAlarmLightToUi((int)AlarmColor::Red);
90 }

93 void AlarmManager::checkoutWheelToDb(WheelDbInfo info)
94 {
95  //检测数据无效
96  if (info.validmatch == 0) {
97  int lastAlarmLevel = getPreCheckstate(info.num);
98  switch (lastAlarmLevel) {
99  case -2:
100  case -1:
101  // 该车轮上一次检测中无检测结果
102  info.alarmlevel = -2;
103  emit showAlarmLightToUi((int)AlarmColor::Yellow);
104  qCritical() << QStringLiteral("本车轮连续两次检测中无有效结果，预警");
105  break;
106  case 0:
107  case 1:
108  case 2:
109  //该车轮上一次检测中有检测结果
110  info.alarmlevel = -1;
111  break;
112  default:
113  break;
114  }
115  }
116  //检测数据有效，对结果进行报警判断
117  else {
118  double error = fabs(info.error);
119  //在预警范围内
120  if (error <= info.warnRatio * 100) {
121  info.alarmlevel = 0;
122  }
123  //预警范围到报警范围之间
124  else if (error <= info.alarmRatio * 100) {
125  int lastAlarmLevel = getPreCheckstate(info.num);
126  switch (lastAlarmLevel) {
127  case -2:
128  case -1:
129  case 0:
130  //上一次结果正常，或者没有有效数据，则预警
131  info.alarmlevel = 1;
132  emit showAlarmLightToUi((int)AlarmColor::Yellow);
133  qCritical() << QStringLiteral("本车轮超出预警值，预警");
134  break;
135  case 1:
136  case 2:
137  //上一次结果是预警或报警，则连续两次超出预警值，报警！
138  info.alarmlevel = 2;
139  emit showAlarmLightToUi((int)AlarmColor::Red);
140  qCritical() << QStringLiteral("本车轮连续两次检测中超出预警值，报警！");
141  break;
142  default:
143  break;
144  }
145  }
146  //超出报警值
147  else {
148  info.alarmlevel = 2;
149  emit showAlarmLightToUi((int)AlarmColor::Red);
150  qCritical() << QStringLiteral("本车轮超出报警值，报警！");
151  }
152  }
153  /* checkstate */
154  switch (info.alarmlevel) {
155  case -2:
156  case 1:
157  case 2:
158  info.checkstate = NeedCheck;
159  break;
160  case -1:
161  case 0:
162  info.checkstate = NoNeedCheck;
163  break;
164  default:
165  break;
166  }
167  //界面显示
168  switch (info.i_o) {
169  case 0:
170  emit showSpeed_o(info.calcspeed);
171  emit showNum_o(info.num);
172  emit showError_o(info.error);
173  break;
174  case 1:
175  emit showSpeed_i(info.calcspeed);
176  emit showNum_i(info.num);
177  emit showError_i(info.error);
178  break;
179  default:
180  break;
181  }
182  if (!insertRecord(info))
183  qDebug() << "insert record failed";
184 }
