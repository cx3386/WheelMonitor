137 void Plc::readSensorPeriodically()
138 {
139  if (!bUsrCtrl)
140  return;
141  QTimer::singleShot(cio0Interval, this, [=]() { readSensorPeriodically(); });
142  auto ansCode = readPLC(genRRCode(0, 1));
143  auto dataList = getRRData(ansCode);
144  if (dataList.size() != 1) // the data should be 1 word
145  {
146  qWarning() << "read cio0 error. anscode " << ansCode;
147  return;
148  }
149  bool ok;
150  int cio0 = QString(dataList.at(0)).toInt(&ok, 16); // convert 'FFFF' to int(10) //0x000~0x1ff, in fact cio0~11(16) ranges to 0xffff.
151  // PATCH: cioO.02换成cio0.10 // [12/5/2018 cx3386]
152  // 获取第i位的数据
153  auto getBit = [](int cio, int loc) -> bool { return cio >> loc & 1; };
154  // 将第i位设置为b
155  auto setBit = [](int& cio, int loc, bool bit) {
156  if (bit)
157  cio |= 1 << loc;
158  else
159  cio &= ~(1 << loc);
160  };
161  auto patch_swicth = [getBit, setBit](int& cio, int bit1, int bit2) {
162  auto bit_tmp = getBit(cio, bit1);
163  setBit(cio, bit1, getBit(cio, bit2));
164  setBit(cio, bit2, bit_tmp);
165  };
166  patch_swicth(cio0, 2, 10);
167  emit cio0Update(cio0);
168  handleSensorDevice[0]->input(cio0);
169  handleSensorDevice[1]->input(cio0);
170 }

172 void Plc::readSpeedPeriodically()
173 {
174  if (!bUsrCtrl)
175  return;
176  QTimer::singleShot(adInterval, this, [=]() { readSpeedPeriodically(); });
177  /* AD分辨率为 1/6000，精度为0.8%满量程（0~55°C），即48单位。
178  4~20mA电流范围对应 0000~1770hex(0~6000)。对应的台车（中轴）速度为0~3.59m/min
179  可转换的数据范围为FED4~189Chex(-300~6300)。3.2~4mA范围内的电流用二进制补码表示。
180  当输入低于指定范围（如低于3.2mA）时，将启动断线检测功能，数据将变为8000hex。
181  注意：这个返回的数据是2byte的有符号数，在**64位编译器**中的short（16bit）符合条件 */
182  const short AD_min = 0xFED4, AD_max = 0x189c, AD_4ma = 0, AD_20ma = 0x1770, AD_err = 0x8000, AD_res = 48;
183  const double scale = 3.59 / 6000; // 1个刻度对应的速度
184 
185  auto ansCode = readPLC(genRRCode(2, 1)); //read cio2 (m+1, AD01) for 1 word
186  auto dataList = getRRData(ansCode);
187  if (dataList.size() != 1) // the data should be 1 word
188  {
189  qWarning() << "read ad error. anscode " << ansCode;
190  return;
191  }
192  bool ok;
193  const short dec = QString(dataList.at(0)).toInt(&ok, 16); // convert 'FFFF' to int(10)
194  QMutexLocker locker(&mutex); // protect truckSpeed
195  truckSpeed = 0;
196  //断线检测
197  if (AD_err == dec) {
198  emit truckSpeedError(1);
199  return;
200  }
201  // out of range(4~20ma), consider the accuracy/resolution
202  if (dec < AD_4ma - AD_res || dec > AD_20ma + AD_res) {
203  emit truckSpeedError(2);
204  return;
205  }
206  //正常数据
207  else {
208  truckSpeed = dec * scale;
209  emit truckSpeedReady();
210  }
211 }

301 bool Plc::checkAnsCode(QByteArray ansCode) const
302 {
303  // null response
304  if (ansCode.isNull()) {
305  return false;
306  }
307  //check the format
308  if (ansCode.startsWith("@00") && ansCode.endsWith("*\r")) {
309  ansCode.chop(2); // chop the terminator
310  auto fcs_recv = ansCode.right(2);
311  ansCode.chop(2); // chop the fcs
312  auto fcs_clc = getFCSCode(ansCode);
313  if (fcs_clc == fcs_recv) {
314  auto headerCode = ansCode.mid(3, 2);
315  auto endCode = ansCode.mid(5, 2);
316  if (endCode == "00") {
317  if (headerCode == "WR" || headerCode == "RR") {
318  return true;
319  }
320  }
321  }
322  }
323  return false;
324 }

331 QByteArrayList Plc::getRRData(QByteArray ansCode) const
332 {
333  QByteArrayList words;
334  if (!checkAnsCode(ansCode)) {
335  qWarning() << "PLCSerial: plc response data err!";
336  return words;
337  }
338  if (ansCode.at(3) != 'R') {
339  qWarning() << "PLCSerial: try to read data from write cmd!";
340  return words;
341  }
342  auto data = ansCode.mid(7, ansCode.size() - 7 - 4);
343  // 读取的数据以字（FFFF）为单位
344  if (data.size() % 4 != 0) {
345  qWarning() << "PLCSerial: plc response data err!";
346  return words;
347  }
348  QByteArray word;
349  for (size_t i = 0; i < data.size(); i++) {
350  word += data.at(i);
351  if ((i + 1) % 4 == 0) {
352  words << word;
353  word.clear();
354  }
355  }
356  return words;
357 }
