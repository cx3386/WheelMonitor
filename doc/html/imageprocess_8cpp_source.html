52  calibratedFrame.copyTo(frameToShow); //here we use deep copy, otherwise we will change calibratedFrame when change frameToshow
53  cvtColor(frameToShow, frameToShow, CV_GRAY2BGR);
54  rectangle(frameToShow, g_imgParam.roiRect, Scalar(0, 0, 255), 2);
55  mutex.unlock();
56  if (!bStopProcess)
57  {
58  //when wheel is stop,
59  //1. stop the process to avoid meaningless count
60  //2. drop the calc results and wait for a come-in signal.
61  //this usually happens when wheel timeout, or the speed is detected as zero
62  if (bWheelStopped)
63  {
64  alarmThisWheel();
65  }
66 
67  else
68  {
69  /*sensor triggered*/
70  //decide if the wheel comes in/out by SENSOR IN/OUT
71  if (g_imgParam.sensorTriggered)
72  {
73  static bool lastIsIn = false; //the wheel is in(1) or out(0) of the detect area. when 1->0(falling edge), which means the wheel is scrolling out, alarm this wheel.
74  //assume wheel is out at initial, won't be set to true until a new wheel comes in.
75  if (bIsInArea) //in detect area
76  {
77  static bool lastCore = false; //record the last return of core process
78  bool nowCore = coreImageProcess(); //0->1, Fragment++ at rise edge. NOTE: it can be 0, which means no circle detected
79  if (nowCore&(!lastCore)) //nowCore == true && lastCore == false
80  {
81  nFragments++;
82  }
83  lastCore = nowCore;
84  }
85  else if (lastIsIn) //lastIsIn==true && bIsInArea==false,which means wheel is out of detect area
86  {
87  //Once a wheel scroll out the detect area, settle it.
88  alarmThisWheel();
89  }
90  lastIsIn = bIsInArea;
91  }
92  /*image triggered*/
93  //for test manually, independent on cameracapsave
94  else
95  {
96  static bool lastCore = false;
97  bool nowCore = coreImageProcess();
98  if (nowCore && !lastCore) //0->1
99  {
100  nFragments++;
101  }
102  else if (!nowCore&&lastCore) //1->0
103  {
104  alarmThisWheel(); //be careful if it cost too much time(>320ms)
105  }
106  lastCore = nowCore;
107  }
108  }
109  }
110  emit imageProcessReady();
111  emit showRealtimeImage();
112 }

248 void ImageProcess::alarmThisWheel()
249 {
250  int totalMatchCount = rtSpeeds.size();
251  int validMatchCount = 0;
252  double meanDiff = 0;
253  double meanRef = 0;
254  int alarmLevel = 0;
255  QString num = QString::fromStdString(ocr->get_final_result());
256  int ocrSize = ocr->size();
257  emit showWheelNum(num);
258  qDebug() << "Num: " << num;
259 
261  if (totalMatchCount < 1)
262  {
263  emit showWheelSpeed(MISS_TEST_SPEED);
264  alarmLevel = -1;
265  qCritical() << "ImageProcess: MISS, no match result";
266  }
268  else
269  {
270  vector<double> rtSpeedDiffs(rtSpeeds.size()); //a vector saves speed difference between imgprocess and PLC(speedAD)
271  transform(rtSpeeds.begin(), rtSpeeds.end(), refSpeeds.begin(), rtSpeedDiffs.begin(), minus<double>()); //rt-ref=diff
272  OutlierHelper outlier;
273  outlier.reject(rtSpeedDiffs); //kick out the bad results by grubbs certain
274  meanRef = outlier.mean(refSpeeds);
275  meanDiff = outlier.mean(rtSpeedDiffs);
276  validMatchCount = rtSpeedDiffs.size();
277 
279  if (validMatchCount >= 10)
280  {
281  double meanRt = meanRef + meanDiff; //mean linear Velocity
282  emit showWheelSpeed(meanRt);
283  qDebug("Speed: %.2lfm/min", meanRt);
285  if (fabs(meanDiff) > meanRef * g_imgParam.warningRatio)
286  {
287  if (fabs(meanDiff) < meanRef * g_imgParam.alarmRatio) //if this wheel is too too slow
288  {
289  emit setAlarmLight(AlarmColorRed);
290  qCritical() << "ImageProcess: WHEEL FATAL ERROR(single too slow)";
291  alarmLevel = 2;
292  }
293  else if (previousAlarmLevel(num) > 0) //if the last wheel is alarmed
294  {
295  emit setAlarmLight(AlarmColorRed);
296  qCritical() << "ImageProcess: WHEEL FATAL ERROR(double slow)";
297  alarmLevel = 2;
298  }
299  else
300  {
301  emit setAlarmLight(AlarmColorYellow);
302  qCritical() << "ImageProcess: WHEEL WARNING ERROR";
303  alarmLevel = 1;
304  }
305  emit showAlarmNum(num);
306  }
308  else
309  {
310  alarmLevel = 0;
311  }
312  }
314  else
315  {
316  if (previousAlarmLevel(num) < 0)
317  {
318  emit setAlarmLight(AlarmColorYellow);
319  qCritical() << "ImageProcess: WHEEL FATAL ERROR(double invalid)";
320  alarmLevel = -2;
321  }
322  else
323  {
324  emit showWheelSpeed(MISS_TEST_SPEED);
325  qCritical() << "ImageProcess: No enough valid results";
326  alarmLevel = -1;
327  }
328  }
329  }
330  if (nFragments != 1)
331  {
332  qWarning() << "ImageProcess: detect count error: " << nFragments;
333  }
334  //when wheel is leaving the detect area
335  QString speedsStr; //" rt1 rt2 rt3 rt4; ref1 ref2 ref3 ref4"
336  for (auto&& sp : rtSpeeds) {
337  speedsStr += QString(" %1").arg(sp, 0, 'f', 2);
338  }
339  speedsStr += ";";
340  for (auto&& sp : refSpeeds) {
341  speedsStr += QString(" %1").arg(sp, 0, 'f', 2);
342  }
343  insertRecord(num, alarmLevel, meanDiff, meanRef, ocrSize, nFragments, totalMatchCount, validMatchCount, speedsStr, HikVideoCapture::videoRelativeFilePath);
344 
345  /************************************************************************/
346  /* reset parameters of this wheel */
347  nFragments = 0;
348  rtSpeeds.clear();
349  refSpeeds.clear();
350  nImgCount = 0; //no need //once settle a wheel, force to zero
351  /************************************************************************/
352 }
