55 void ImageProcess::handleFrame()
56 {
57  makeFrame4Show(); //只要cap发送信号，则显示到界面上
58  if (bUsrCtrl) {
59  // 台车停止，立马结算当前车轮，并停止处理。
60  // 发生在光电开关未在规定时间内检测到离开信号，或者测到台车速度为0
61  if (bIsTruckStopped) {
62  if (!checkoutAlready)
63  {
64  checkoutWheel();
65  checkoutAlready = true;
66  }
67  }
68  else {
69  //sensor triggered. 通过SENSOR IN/OUT信号开始，注意，1个周期只收到1次in，1次out。/结束本车轮的检测，而无论core的返回值
70  if (imProfile->sensorTriggered) {
71  //int state = _DZRecorder.state(0);
72  // 车轮进入、正在DZ
73  //if (state == LevelRecorder::HighLevel || state == LevelRecorder::PositiveEdge) //in detect area
74  if (bInDz)
75  {
76  checkoutAlready = false;
77  int nCore = coreImageProcess(); //0->1, Fragment++ at rise edge. NOTE: it can be 0, which means no circle detected
78  if (nCore == LocateFail && nCore_pre > RMA) //nowCore == true && lastCore == false
79  interrupts++;
80  nCore_pre = nCore;
81  }
82  // 车轮离开DZ，结算
83  else {
84  if (!checkoutAlready)
85  {
86  checkoutWheel();
87  checkoutAlready = true;
88  }
89  }
90  }
91  //image triggered
92  //通过LMA变InMA的上升沿开始，InMA变RMA的下降沿结算车轮；
93  // 边界反复跳动：开始1(跳过)-->开始2,3..（清零）-->中断-->结束1（结算）-->结束2,3..（跳过）
94  else {
95  int nCore = coreImageProcess();
96  // LMA->InMA上升沿 开始
97  if (nCore > RMA && nCore_pre == LMA) {
98  // 开始1
99  if (_MAState != 0) {
100  _MAState = 0;
101  }
102  // 开始2,3...
103  else {
104  rtSpeeds.clear();
105  refSpeeds.clear();
106  interrupts = 0;
107  //wheelFrame_pre.release(); // 因为pre是LMA,所以frame_pre必然是空的
108  }
109  }
110  // InMA->RMA下降沿 结算
111  else if (nCore == RMA && nCore_pre > RMA) {
112  // 结束1
113  if (_MAState != 1) {
114  _MAState = 1;
115  // ncore重置为LocateFail，实际应该为RMA, 但并不影响判断（无pre==RMA或FindFail的判断条件）
116  // 此时wheelFrame_pre已经自动release了
117  checkoutWheel();
118  }
119  }
120  // InMA->MissWheel 中断
121  else if (nCore == LocateFail && nCore_pre > RMA) {
122  _MAState = 2;
123  interrupts++;
124  }
125  nCore_pre = nCore;
126  }
127  }
128  }
129  emit frameHandled();
130  emit showFrame();
131 }

143 int ImageProcess::coreImageProcess()
144 {
145  ocr->core_ocr(srcImg); //card num detect gzy 2017/11/9
146  Mat monitorArea = undistortedFrame(imProfile->roiRect); //截取监测区域
147  equalizeHist(monitorArea, monitorArea); //equalize the image, this will change calibratedFrame//2017/12/21
148  Mat blurImage; //use blur image for hough circles, use source image for matches
149 
150  // Reduce the noise to avoid false circle detection
151  GaussianBlur(monitorArea, blurImage, Size(imProfile->gs1, imProfile->gs1), 2, 2); //the greater the kernel size is, the less time the hough cost
152  // 霍夫匹配，寻找车轮的外圆。注意：该方法获取的圆可能部分在图像外面
153  vector<Vec3f> circles;
154  int max_r = std::min(monitorArea.rows / 2, imProfile->radius_max);// 防止纵向超出：通常高度比较小
155  HoughCircles(blurImage, circles, CV_HOUGH_GRADIENT, imProfile->dp, imProfile->minDist, imProfile->param1, imProfile->param2, imProfile->radius_min, max_r);
156  //无霍夫圆
157  if (circles.empty()) {
158  wheelFrame_pre.release(); // 有效帧将不连续，置空前一帧
159  return LocateFail; // 车轮定位丢失
160  }
161 
162  //取霍夫圆的最大一个，当作车轮外径。取外切矩形用于匹配
163  //patch：唯一一个霍夫圆
164  Point center0(cvRound(circles[0][0]), cvRound(circles[0][1])); // 圆心
165  int radius = cvRound(circles[0][2]); // 半径
166  Rect wheelRect((center0.x - radius), (center0.y - radius), 2 * radius, 2 * radius); // 外切矩形的位置
167 
168  //判断车轮是否在监测区域内
169  Rect monitorRect(0, 0, monitorArea.cols, monitorArea.rows);
170  bool bInside = isInside(wheelRect, monitorRect);
171 
172  //draw circle to UI
173  auto circleColor = bInside ? Scalar(0, 255, 0) : Scalar(0, 255, 255); // 如果在里面则画绿，否则画黄
174  Size size;
175  Point ofs;
176  monitorArea.locateROI(size, ofs); // get the offset of monitorArea from the srcImg
177  mutex.lock();
178  circle(frameToShow, center0 + ofs, 3, circleColor, -1); // center
179  circle(frameToShow, center0 + ofs, radius, circleColor, 1, LINE_AA); // round
180  mutex.unlock();
181 
182  // 未完全在监测区域内，调用时应注意颤抖引起的在边界的反复横跳
183  if (!bInside) {
184  //车轮在MA的左边，即未进入
185  if (wheelRect.tl().x < monitorRect.tl().x) {
186  wheelFrame_pre.release(); // 置空前一帧
187  return LMA;
188  }
189  //车轮在MA的右侧，即已退出
190  if (wheelRect.br().x > monitorRect.br().x) {
191  wheelFrame_pre.release(); // 置空前一帧
192  return RMA; //软件判别法应该结算该车轮
193  }
194  //车轮超出MA的上下界
195  if (wheelRect.tl().y < monitorRect.tl().y || wheelRect.br().y > monitorRect.br().y) {
196  wheelRect = wheelRect & monitorRect;//取交集
197  }
198  }
199 
200  // 至此成功锁定（区域内的）车轮，保存
201  Mat matchSrc1 = wheelFrame_pre;
202  Mat matchSrc2 = monitorArea(wheelRect); //外切矩形图像,注意：wheelrect必须在monitor内
203  wheelFrame_pre = matchSrc2;
204 
205  // 前一帧为空
206  if (matchSrc1.empty()) {
207  return NoPre; // 等待匹配帧
208  }
209 
210  // 开始匹配 //orb matches
211  Mat image_matches; // show in UI
212  double oneAngle;
213  if (!rMatcher->match(matchSrc1, matchSrc2, configHelper->matchMaskOuterRatio, configHelper->matchMaskInnerRatio, image_matches, oneAngle)) //200ms
214  {
215  //qDebug() << "match fail";
216  return MatchFail;
217  }
218  mutex.lock();
219  image_matches.copyTo(matchResult);
220  mutex.unlock();
221  emit showMatch();
222  //angle to linear velocity
223  double rtSpeed = oneAngle * imProfile->angle2Speed();
224  qDebug("dev[%d] Speed: %.2lf; ref: %.2lf", deviceIndex, rtSpeed, rtRefSpeed);
225  rtSpeeds.push_back(rtSpeed);
226  refSpeeds.push_back(rtRefSpeed);
227 
228  // if truck stops, give up this wheel.
229  // to avoid the error rtSpeed, this judge should be refSpeed and rtSpeed
230  // this won't stop the video cap save, until save timeout(100s)
231  if (rtRefSpeed < 0.05 && rtSpeed < 0.05) {
232  bIsTruckStopped = true;
233  return TruckSpZero;
234  }
235  return Success;
236 }

249 void ImageProcess::checkoutWheel()
250 {
251  /* 未在此处初始化的wheelDbInfo, 只在特定时候（valid）时候才有效，refspeed为coreProcess时取得的 */
252  WheelDbInfo wheelDbInfo;
253  wheelDbInfo.i_o = deviceIndex;
254  wheelDbInfo.num = QString::fromStdString(ocr->get_final_result());
255  // 记录所有的速度值
256  wheelDbInfo.speeds = genSpeeds(rtSpeeds, refSpeeds);
257  wheelDbInfo.totalmatch = rtSpeeds.size(); // match size
258  //如果有足够多数据，进行计算
259  if (refSpeeds.size() >= IM_PROC_VALID_MIN_COUNT) {
260  //计算每一次数据的误差：rt-ref=diff
261  vector<double> absErrors(rtSpeeds.size());
262  transform(rtSpeeds.begin(), rtSpeeds.end(), refSpeeds.begin(), absErrors.begin(), minus<double>());
263  // 剔除异常。注意：总会有结果
264  OutlierHelper outlier;
265  outlier.removeOutliers(absErrors);
266  double refspeed = outlier.mean(refSpeeds);
267  double absError = outlier.mean(absErrors);
268  wheelDbInfo.validmatch = absErrors.size();
269  // 写入数据库的值
270  wheelDbInfo.calcspeed = QString::number(absError + refspeed, 'f', 2).toDouble();
271  wheelDbInfo.refspeed = QString::number(refspeed, 'f', 2).toDouble();
272  wheelDbInfo.error = QString::number(absError / refspeed * 100, 'f', 1).toDouble();
273  qDebug() << "Speed: " << wheelDbInfo.calcspeed << ";Ref: " << wheelDbInfo.refspeed;
274  }
275  //无有效数据
276  else
277  {
278  wheelDbInfo.validmatch = 0;
279  wheelDbInfo.calcspeed = IM_PROC_INVALID_SPEED; // 计算值为溢出
280  wheelDbInfo.refspeed = rtRefSpeed; // 当前参考速度
281  wheelDbInfo.error = 0.0;
282  if (wheelDbInfo.totalmatch == 0)
283  {
284  qCritical() << QStringLiteral("无匹配结果");
285  }
286  else
287  {
288  qCritical() << QStringLiteral("匹配结果太少");
289  }
290  }
291  wheelDbInfo.time = QDateTime::currentDateTime().toString("yyyy-MM-dd hh:mm:ss");
292  wheelDbInfo.ocrsize = ocr->size();
293  wheelDbInfo.fragment = interrupts;
294 
295  wheelDbInfo.videopath = videoCapture->getVideoRelativeFilePath();
296  wheelDbInfo.warnRatio = imProfile->warningRatio;
297  wheelDbInfo.alarmRatio = imProfile->alarmRatio;
298 
299  emit wheelNeedHandled(wheelDbInfo);
300  /* reset parameters of this wheel */
301  clearWheel();
302 }
