156  int h = in.rows;
157  int w = in.cols;
158  Mat transformMat = Mat::eye(2, 3, CV_32F);
159  int m = max(w, h);
160  transformMat.at<float>(0, 2) = m / 2 - w / 2;
161  transformMat.at<float>(1, 2) = m / 2 - h / 2;

197 float OCR::oudistance(Mat a, Mat b) const {
198  float distance = 0;
199  for (int i = 0; i < a.cols; i++) {
200  for (int j = 0; j < a.rows; j++) {
201  distance += (a.at<uchar>(i, j) - b.at<uchar>(i, j))*(a.at<uchar>(i, j) - b.at<uchar>(i, j));
202  }
203  }
204  return sqrt(distance);
205 }
