170 float OCR::oudistance(Mat a, Mat b) const
171 {
172  float distance = 0;
173  for (int i = 0; i < a.cols; i++) {
174  for (int j = 0; j < a.rows; j++) {
175  distance += (a.at<uchar>(i, j) - b.at<uchar>(i, j)) * (a.at<uchar>(i, j) - b.at<uchar>(i, j));
176  }
177  }
178  return sqrt(distance);
179 }
