176  switch (brokens)

179  case 1:

181  updateSensor = false;

183  break;
184  case 3://坏了3个了，最后一个不亮，可能是设备坏了，也可能是掉轮
185  updateSensor = true;

187  break;

192  break;

194  }

196  if (updateSensor)
