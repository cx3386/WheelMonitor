120 void HikVideoCapture::startRecord()
121 {
122  QMutexLocker locker(&mutex);
123  if (bIsRecording)
124  return;
125  auto date = QDate::currentDate().toString("yyyyMMdd");
126  auto dateTime = QDateTime::currentDateTime().toString("yyyyMMddhhmmss");
127  auto mark = getDeviceMark(deviceIndex);
128  videoRelativeFilePath = QStringLiteral("%1/%2_%3.mp4").arg(date).arg(mark).arg(dateTime);
129  QString videoFilePath = QStringLiteral("%1/%2").arg(videoDirPath).arg(videoRelativeFilePath);
130  QByteArray ba = videoFilePath.toLocal8Bit();
131  if (!NET_DVR_SaveRealData(lRealPlayHandle_SD, ba.data())) {
132  qDebug() << "HikVideoCapture: startRecord error";
133  return;
134  }
135  // 子线程操作
136  timeoutTimer = new QTimer;
137  timeoutTimer->setSingleShot(true); // 不直接用QTimer::singleShot，因为一旦开始，无法stop
138  timeoutTimer->setInterval(MAX_RECORD_MSEC); // 设置超时时间为100s
139  connect(timeoutTimer, &QTimer::timeout, this, [=] {
140  stopRecord(); //子线程
141  emit recordTimeout();
142  });
143  timeoutTimer->start(); //If the timer is already running, it will be stopped and restarted.
144  bIsRecording = true;
145  emit recordON();
146 }

204 void CALLBACK HikVideoCapture::fRealDataCallBack(LONG lRealHandle, DWORD dwDataType, BYTE* pBuffer, DWORD dwBufSize, void* pUser) //void *pUser
205 {
206  auto* pCap = static_cast<HikVideoCapture*>(pUser);
207  LONG& nPort = pCap->nPort;
208  auto DecCBFun = [pCap]() -> auto
209  {
210  int index = pCap->deviceIndex;
211  switch (index) {
212  case 0:
213  HikVideoCapture::pCap0 = pCap;
214  return HikVideoCapture::DecCBFun0;
215  case 1:
216  HikVideoCapture::pCap1 = pCap;
217  return HikVideoCapture::DecCBFun1;
218  default:
219  break;
220  }
221  };
222  switch (dwDataType) {
223  case NET_DVR_SYSHEAD: //系统头
224 
225  if (!PlayM4_GetPort(&nPort)) { //获取播放库未使用的通道号
226  break;
227  }
228  //m_iPort = lPort; //第一次回调的是系统头，将获取的播放库port号赋值给全局port，下次回调数据时即使用此port号播放
229  if (dwBufSize > 0) {
230  if (!PlayM4_SetStreamOpenMode(nPort, STREAME_REALTIME)) { //设置实时流播放模式
231  break;
232  }
233 
234  if (!PlayM4_OpenStream(nPort, pBuffer, dwBufSize, 10 * 1024 * 1024)) { //打开流接口
235  break;
236  }
237 
238  if (!PlayM4_Play(nPort, nullptr)) { //解码，不播放
239  break;
240  }
241  if (!PlayM4_SetDecCallBack(nPort, DecCBFun())) {
242  break;
243  }
244  }
245  break;
246  case NET_DVR_STREAMDATA: //码流数据
247  if (dwBufSize > 0 && nPort != -1) {
248  if (!PlayM4_InputData(nPort, pBuffer, dwBufSize)) {
249  qDebug() << "HikVideoCapture: fRealDataCallBack error" << PlayM4_GetLastError(nPort);
250  break;
251  }
252  }
253  break;
254  default: //其他数据
255  if (dwBufSize > 0 && nPort != -1) {
256  if (!PlayM4_InputData(nPort, pBuffer, dwBufSize)) {
257  break;
258  }
259  }
260  break;
261  }
262 }
