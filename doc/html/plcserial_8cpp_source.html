136 {
137  QMutexLocker locker(&mutex);
138 
139  QByteArray plcData = READ_SENSOR_STATE;
140  plcSerialPort->write(plcData);
141  if (plcSerialPort->waitForBytesWritten(100))
142  {
143  if (plcSerialPort->waitForReadyRead(100))
144  {
145  QByteArray responseData = plcSerialPort->readAll();
146  while (plcSerialPort->waitForReadyRead(100))
147  responseData += plcSerialPort->readAll();
148 
149  if (responseData == SENSOR_L_OFF_R_OFF)
150  {//00
151  if (sensorRight == true && sensorLeft == false)
152  {//01->00 wheel leaves the right sensor
153  emit sensorOUT();
154  }
155  sensorRight = false;
156  sensorLeft = false;
157  }
158  else if (responseData == SENSOR_L_OFF_R_ON)
159  {//01
160  sensorRight = true;
161  sensorLeft = false;
162  }
163  else if (responseData == SENSOR_L_ON_R_OFF)
164  {//10
165  if (sensorRight == false && sensorLeft == false)
166  {//00->10 wheel enters the left sensor
167  emit sensorIN();
168  }
169  sensorRight = false;
170  sensorLeft = true;
171  }
172  else if (responseData == SENSOR_L_ON_R_ON)
173  {//11
174  qWarning() << "PLC(sensor): Both sensors are triggered at the same time, please check if the sensor is broken";
175  sensorRight = true;
176  sensorLeft = true;
177  }
178  else
179  {//wrong response
180  qWarning() << "PLC(sensor): Wrong response";
181  }
182  }
183  else
184  {
185  //QString str(tr("Wait read response timeout %1")
186  //.arg(QTime::currentTime().toString()));
187  qWarning() << "PLC(sensor): Wait read response timeout";
188  }
189  }
190  else
191  {
192  qWarning() << "PLC(sensor): Wait write request timeout";
193  }
194 }

196 void PLCSerial::readAD() //write and read one time need at least 300ms
197 {
198  QMutexLocker locker(&mutex);
199  QByteArray plcData = READ_AD;
200  plcSerialPort->write(plcData);
201  if (plcSerialPort->waitForBytesWritten(100))
202  {
203  if (plcSerialPort->waitForReadyRead(100))
204  {
205  QByteArray responseData = plcSerialPort->readAll();
206  while (plcSerialPort->waitForReadyRead(100))
207  responseData += plcSerialPort->readAll();
208  mutex.unlock();
209  bool ok;
210  int tmpDec = QString(responseData.mid(7, 4)).toInt(&ok, 16);
211  if (8000 == tmpDec || 65236 <= tmpDec || 65531 <= tmpDec)//8000 or under -5, think ad module is down
212  {
213  emit ADdisconnected();
214  mutex.lock();
215  speedAD = 0;
216  }
217  else
218  {
219  mutex.lock();

222  }
223  }
